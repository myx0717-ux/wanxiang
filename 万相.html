function drawEdges() {

  const svg = document.getElementById('canvas-svg');

  if(!svg) return;

  let svgHtml = `<defs><marker id="arrow" viewBox="0 0 10 10" refX="9" refY="5" markerWidth="6" markerHeight="6" orient="auto-start-reverse"><path d="M 0 0 L 10 5 L 0 10 z" fill="#9ca3af" /></marker></defs>`;

  

  EDGES.forEach(e => {

    const fromNode = NODES.find(n => n.id === e.from);

    const toNode = NODES.find(n => n.id === e.to);

    if(fromNode && toNode) {

      const startX = fromNode.x + 110; 

      const startY = fromNode.y + 110; 

      const endX = toNode.x + 110;

      const endY = toNode.y;

      const curve = `M ${startX} ${startY} C ${startX} ${startY + 30}, ${endX} ${endY - 30}, ${endX} ${endY}`;

      svgHtml += `<path d="${curve}" fill="none" stroke="#9ca3af" stroke-width="2" marker-end="url(#arrow)" />`;

    }

  });

  svg.innerHTML = svgHtml;

}


function startDrag(e, id) {

  if (e.target.classList.contains('fn-port')) return;

  draggedNode = NODES.find(n => n.id === id);

  if(!draggedNode) return;

  const rect = document.getElementById(id).getBoundingClientRect();

  dragOffset.x = e.clientX - rect.left;

  dragOffset.y = e.clientY - rect.top;

  document.addEventListener('mousemove', drag);

  document.addEventListener('mouseup', endDrag);

}


function drag(e) {

  if (!draggedNode) return;

  const canvasRect = document.getElementById('tc-canvas').getBoundingClientRect();

  let newX = e.clientX - canvasRect.left - dragOffset.x;

  let newY = e.clientY - canvasRect.top - dragOffset.y;

  draggedNode.x = newX;

  draggedNode.y = newY;

  const el = document.getElementById(draggedNode.id);

  el.style.left = newX + 'px';

  el.style.top = newY + 'px';

  drawEdges();

}


function endDrag() {

  draggedNode = null;

  document.removeEventListener('mousemove', drag);

  document.removeEventListener('mouseup', endDrag);

}
